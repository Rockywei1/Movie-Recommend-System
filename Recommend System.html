# Get the director's name from the crew feature. If director is not listed, return NaN
def get_director(x):
    for i in x:
        if i['job'] == 'Director':
            return i['name']
    return np.nan

# Returns the list top 3 elements or entire list; whichever is more.
def get_list(x):
    if isinstance(x, list):
        names = [i['name'] for i in x]
        #Check if more than 3 elements exist. If yes, return only first three. If no, return entire list.
        if len(names) > 5:
            names = names[:5]
        return names

    #Return empty list in case of missing/malformed data
    return []

def get_most_10(data, column):
    temp_data = data.apply(lambda x: pd.Series(x[column], dtype='object'),axis=1).stack().reset_index(level=1, drop=True)
    a = temp_data.value_counts().reset_index()
    a.rename(columns={0: 'Value'})
    fig = px.bar(a.iloc[:10][::-1], x=0, y='index', orientation='h', title="Top 10 {}".format(column),color_discrete_sequence=px.colors.qualitative.G10)
    fig.show()

def get_recommendations(movie_title, cosine_similarity = cosine_sim):
    # Get the index corresponding to original_title

    index_movie = titles[movie_title]
    
    similarities = cosine_similarity[index_movie]
    
    # Get the pairwsie similarity scores

    similarity_scores = list(enumerate(similarities))
    
    # Sort the movies

    similarity_scores = sorted(similarity_scores , key=lambda x: x[1], reverse = True)
    
    # Scores of the 10 most similar movies

    similarity_scores = similarity_scores[1:11]
    
    similar_indexes = [x[0] for x in similarity_scores]
    
    print("if you like {}, maybe you'll enjoy: \n\n".format(movie_title))
    
    return df2['title_x'].iloc[similar_indexes]

Summary¶

Based on the above results, it can be seen that the content-based filtering system also has some limitations:¶

1. When users search for movies similar to Spy Kids, the recommendation system may select movies that users think are not similar. To improve the system, consider replacing TF-IDF algorithm with word counting and calculating other similar scores.¶

2. The system will only analyze the overview of each film. If we want to analyze other characteristics, such as actors, directors, genres, etc. like Ibtesam, we need to make improvements in finding similar films.¶

3. At present, this system can only give recommendations based on the relevance of film characteristics. This leaves out films from other genres that users might be interested in. We need to use a collaborative filtering system to solve this problem, but the current database does not contain user information.¶
